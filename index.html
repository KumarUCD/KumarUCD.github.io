## Import data
res <- read.csv("DE_gene_list.csv", header=TRUE)
## Make another data set for lebeling: method a) or  b)
resdata <- subset(res, Gene != "")
#resdata<-res[!(is.na(res$Gene) | res$Gene==""), ]
## Can be upload saperate file
#resdata <- read.csv("res-diffexpr-results_IMD4vsIMSL_steven_v3.csv", header=TRUE)
#resdata2 <- read.csv("res-diffexpr-results_IMD4vsIMSL_steven_v3.csv", header=TRUE)
volcanoplot_1 <- function (res, lfcthresh=1, sigthresh=0.05, main="Volcano Plot", labelsig=TRUE, textcx=1, ...) {
  with(res, plot(log2FoldChange, -log10(padj), pch=20, col="black",main="Volcano Plot", cex=1.5, xlab=bquote(~Log[2]~fold~change), ylab=bquote(~-log[10]~adj ~p~value)))
  #with(res, plot(log2FoldChange, -log10(padj), pch=20, col="white",main="Volcano Plot", cex=1.5, xlab=bquote(~Log[2]~fold~change), ylab=bquote(~-log[10]~adj ~p~value)))
  with(subset(res, padj<0.05 & log2FoldChange>=1), points(log2FoldChange, -log10(padj), pch=20, col="red", cex=1.0))
  #with(subset(res, padj<0.05 & log2FoldChange>=1), points(log2FoldChange, -log10(padj), pch=20, col="white", cex=1.0))
  with(subset(res, padj<0.05 & log2FoldChange<=-1), points(log2FoldChange, -log10(padj), pch=20, col="lightslateblue", cex=1.0))
  #with(subset(res, padj<0.05 & log2FoldChange<=-1), points(log2FoldChange, -log10(padj), pch=20, col="white", cex=1.0))
  #abline(v=0, col="red", lty=1, lwd=5.0)
  abline(v=-1, col="blue", lty=2, lwd=3.0)
  abline(v=1, col="red", lty=2, lwd=3.0)
  abline(h=1.3, col="black", lty=1, lwd=2.0)
  if (labelsig) {
    require(calibrate)
    ## For significant genes only
    #with(subset(res, padj<sigthresh & abs(log2FoldChange)), textxy(log2FoldChange, -log10(padj), labs=Gene, cex=1, offset=.8))
    ## for all listed genes
    with(subset(res), textxy(log2FoldChange, -log10(padj), labs=Gene, col="black", cex=1, offset=.8))
    ## For significant genes only
    #with(subset(resdata, padj<sigthresh & abs(log2FoldChange)), points(log2FoldChange, -log10(padj), pch=1, col="black", lwd=3, cex=2.0))
    ## for all listed genes
    with(subset(resdata), points(log2FoldChange, -log10(padj), pch=1, col="black", lwd=3, cex=2.0))
  }
}
## For instant view
volcanoplot_1(res, textcx=1, xlim=c(-12, 15))
## For saving high resolution image
png("Final_volcano_plot_1.png", 1200, 1000, pointsize=20)
volcanoplot_1(res, lfcthresh=1, sigthresh=0.05, textcx=1, xlim=c(-12, 15))
dev.off()
